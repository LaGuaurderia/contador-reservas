  <script>
    async function obtenerReservas() {
      try {
        const response = await fetch('https://docs.google.com/spreadsheets/d/e/2PACX-1vT_nn_ZSKMibu0CZRoaRXzFyXepjJG2bjaT1H-s35ns3iZuIrVkU76h5hlI1uds8Q/pub?gid=1717728688&single=true&output=csv');
        const data = await response.text();
        const filas = data.split('\n');
        const totalReservas = filas.length - 1;
        document.getElementById('contador').textContent = totalReservas;
      } catch (error) {
        document.getElementById('contador').textContent = 'Error';
      }
    }
    obtenerReservas();
  </script>
